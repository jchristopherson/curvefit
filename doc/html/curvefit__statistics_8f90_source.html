936  real(real64), intent(in) :: a, x
937  real(real64) :: g

940  integer(int32), parameter :: itmax = 500

943  real(real64), parameter :: two = 2.0d0
944 
945  ! Local Variables
946  integer(int32) :: i
947  real(real64) :: an, b, c, d, del, h, gln, eps, fpmin
948 
949  ! Initialization
950  eps = epsilon(eps)
951  fpmin = tiny(fpmin) / eps
952  gln = log_gamma(a)
953  b = x + one - a
954  c = one / fpmin
955  d = one / b
956  h = d
957  g = zero
958  do i = 1, itmax
959  an = -i * (i - a)
960  b = b + two
961  d = an * d + b
962  if (abs(d) < fpmin) d = fpmin
963  c = b + an / c
964  if (abs(c) < fpmin) c = fpmin
965  d = one / d
966  del = d * c
967  h = h * del
968  if (abs(del - one) < eps) then
969  g = exp(-x + a * log(x) - gln) * h
970  return
971  end if
972  end do
