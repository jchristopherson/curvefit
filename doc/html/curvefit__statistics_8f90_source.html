1042  ! Arguments
1043  real(real64), intent(in) :: a, x
1044  real(real64) :: g
1045 
1046  ! Parameters
1047  integer(int32), parameter :: itmax = 500
1048  real(real64), parameter :: zero = 0.0d0
1049  real(real64), parameter :: one = 1.0d0
1050  real(real64), parameter :: two = 2.0d0
1051 
1052  ! Local Variables
1053  integer(int32) :: i
1054  real(real64) :: an, b, c, d, del, h, gln, eps, fpmin
1055 
1056  ! Initialization
1057  eps = epsilon(eps)
1058  fpmin = tiny(fpmin) / eps
1059  gln = log_gamma(a)
1060  b = x + one - a
1061  c = one / fpmin
1062  d = one / b
1063  h = d
1064  g = zero
1065  do i = 1, itmax
1066  an = -i * (i - a)
1067  b = b + two
1068  d = an * d + b
1069  if (abs(d) < fpmin) d = fpmin
1070  c = b + an / c
1071  if (abs(c) < fpmin) c = fpmin
1072  d = one / d
1073  del = d * c
1074  h = h * del
1075  if (abs(del - one) < eps) then
1076  g = exp(-x + a * log(x) - gln) * h
1077  return
1078  end if
1079  end do
