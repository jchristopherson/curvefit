262  function im_locate(this, pt, err) result(j)
263  ! Arguments
264  class(interp_manager), intent(inout) :: this
265  real(real64), intent(in) :: pt
266  class(errors), intent(inout), optional, target :: err
267  integer :: j
268 
269  ! Local Variables
270  integer(int32) :: n, m, jhi, jmid, jlo
271  logical :: ascnd
272  class(errors), pointer :: errmgr
273  type(errors), target :: deferr
274 
275  ! Initialization
276  j = 0
277  n = size(this%m_x)
278  m = this%m_order + 1
279  ascnd = this%m_x(n) >= this%m_x(1)
280  jlo = 1
281  jhi = n
282  if (present(err)) then
283  errmgr => err
284  else
285  errmgr => deferr
286  end if
287 
288  ! Ensure data has been defined
289  if (.not.allocated(this%m_x) .or. .not.allocated(this%m_y)) then
290  call errmgr%report_error("im_locate", "No data has been defined.", &
291  cf_no_data_defined_error)
292  return
293  end if
294 
295  ! Process
296  do while (jhi - jlo > 1)
297  jmid = (jhi + jlo) / 2
298  if (pt >= this%m_x(jmid) .eqv. ascnd) then
299  jlo = jmid
300  else
301  jhi = jmid
302  end if
303  end do
304 
305  ! Check to see if we should use a more efficient search approach next
306  ! time
307  this%m_correlated = abs(jlo - this%m_savedIndex) <= this%m_indexCheck
308  this%m_savedIndex = jlo
309 
310  ! Output
311  ! j = max(1, min(n + 1 - m, jlo - (m - 1) / 2))
312  if (pt == this%m_x(1)) then
313  j = 1
314  else if (pt == this%m_x(n)) then
315  j = n - 1
316  else
317  j = jlo
318  end if

337  function im_hunt(this, pt, err) result(j)
338  ! Arguments
339  class(interp_manager), intent(inout) :: this
340  real(real64), intent(in) :: pt
341  class(errors), intent(inout), optional, target :: err
342  integer(int32) :: j
343 
344  ! Local Variables
345  integer(int32) :: jlo, jmid, jhi, inc, n, m
346  logical :: ascnd
347  class(errors), pointer :: errmgr
348  type(errors), target :: deferr
349 
350 
351  ! Initialization
352  j = 0
353  n = size(this%m_x)
354  m = this%m_order + 1
355  jlo = this%m_savedIndex
356  inc = 1
357  ascnd = this%m_x(n) > this%m_x(1)
358  if (present(err)) then
359  errmgr => err
360  else
361  errmgr => deferr
362  end if
363 
364  ! Ensure data has been defined
365  if (.not.allocated(this%m_x) .or. .not.allocated(this%m_y)) then
366  call errmgr%report_error("im_hunt", "No data has been defined.", &
367  cf_no_data_defined_error)
368  return
369  end if
370 
371  ! Process
372  if (jlo < 1 .or. jlo > n) then
373  jlo = 1
374  jhi = n
375  else
376  if (pt >= this%m_x(jlo) .eqv. ascnd) then
377  do
378  jhi = jlo + inc
379  if (jhi >= n) then
380  jhi = n
381  exit
382  else if (pt < this%m_x(jhi) .eqv. ascnd) then
383  exit
384  else
385  jlo = jhi
386  inc = inc + inc
387  end if
388  end do
389  else
390  jhi = jlo
391  do
392  jlo = jlo - inc
393  if (jlo <= 1) then
394  jlo = 1
395  exit
396  else if (pt >= this%m_x(jlo) .eqv. ascnd) then
397  exit
398  else
399  jhi = jlo
400  inc = inc + inc
401  end if
402  end do
403  end if
404  end if
405 
406  ! The hunt is done, so begin the final bisection phase
407  do while (jhi - jlo > 1)
408  jmid = (jhi + jlo) / 2
409  if (pt >= this%m_x(jmid) .eqv. ascnd) then
410  jlo = jmid
411  else
412  jhi = jmid
413  end if
414  end do
415 
416  ! Check to see if we should hunt or locate the next time around
417  this%m_correlated = abs(jlo - this%m_savedIndex) <= this%m_indexCheck
418  this%m_savedIndex = jlo
419 
420  ! Output
421  ! j = max(1, min(n + 1 - m, jlo - (m - 1) / 2))
422  if (pt == this%m_x(1)) then
423  j = 1
424  else if (pt == this%m_x(n)) then
425  j = n - 1
426  else
427  j = jlo
428  end if
429  end function

649  function pi_raw_interp(this, jlo, pt) result(yy)
650  ! Arguments
651  class(polynomial_interp), intent(inout) :: this
652  integer(int32), intent(in) :: jlo
653  real(real64), intent(in) :: pt
654  real(real64) :: yy
655 
656  ! Local Variables
657  integer(int32) :: i, ind, m, ns, mm, jl
658  real(real64) :: den, dif, dift, ho, hp, w
659 
660  ! Initialization
661  mm = this%m_order + 1
662  ns = 1
663  jl = jlo - 1
664  dif = abs(pt - this%m_x(jl + 1))
665 
666  ! Find the index NS of the closest table entry, and then initialize
667  ! the C and D arrays.
668  do i = 1, mm
669  ind = jl + i
670  dift = abs(pt - this%m_x(ind))
671  if (dift < dif) then
672  ns = i
673  dif = dift
674  end if
675  this%m_c(i) = this%m_y(ind)
676  this%m_d(i) = this%m_y(ind)
677  end do
678 
679  ! Define the initial approximation to the interpolated point
680  yy = this%m_y(jl + ns)
681  ns = ns - 1
682 
683  ! Build the tables, and define the interpolated point
684  do m = 1, mm-1
685  do i = 1, mm - m
686  ind = jl + i
687  ho = this%m_x(ind) - pt
688  hp = this%m_x(ind+m) - pt
689  w = this%m_c(i+1) - this%m_d(i)
690  den = ho - hp
691  den = w / den
692  this%m_d(i) = hp * den
693  this%m_c(i) = ho * den
694  end do
695  if (2 * ns < mm - m) then
696  this%m_dy = this%m_c(ns + 1)
697  else
698  this%m_dy = this%m_d(ns)
699  ns = ns - 1
700  end if
701  yy = yy + this%m_dy
702  end do
703  end function

732  subroutine penta_solve(a1, a2, a3, a4, a5, b, x)
733  ! Arguments
734  real(real64), intent(in), dimension(:) :: a1, a5
735  real(real64), intent(inout), dimension(:) :: a2, a3, a4, b
736  real(real64), intent(out), dimension(:) :: x
737 
738  ! Local Variables
739  integer(int32) :: i, n
740  real(real64) :: xmult
741 
742  ! Initialization
743  n = size(a1)
744 
745  ! Process
746  do i = 2, n - 1
747  xmult = a2(i) / a3(i - 1)
748  a3(i) = a3(i) - xmult * a4(i - 1)
749  a4(i) = a4(i) - xmult * a5(i - 1)
750  b(i) = b(i) - xmult * b(i - 1)
751  xmult = a1(i + 1) - xmult * a4(i - 1)
752  a2(i + 1) = a2(i + 1) - xmult * a4(i - 1)
753  a3(i + 1) = a3(i + 1) - xmult * a5(i - 1)
754  b(i + 1) = b(i + 1) - xmult * b(i - 1)
755  end do
756 
757  xmult = a2(n) / a3(n - 1)
758  a3(n) = a3(n) - xmult * a4(n - 1)
759  x(n) = (b(n) - xmult * b(n - 1)) / a3(n)
760  x(n - 1) = (b(n - 1) - a4(n - 1) * x(n)) / a3(n - 1)
761  do i = n - 2, 1, -1
762  x(i) = (b(i) - a4(i) * x(i + 1) - a5(i) * x(i + 2)) / a3(i)
763  end do
764  end subroutine

774  function si_raw_interp(this, jlo, pt) result(yy)
775  ! Arguments
776  class(spline_interp), intent(inout) :: this
777  integer(int32), intent(in) :: jlo
778  real(real64), intent(in) :: pt
779  real(real64) :: yy
780 
781  ! Parameters
782  real(real64), parameter :: half = 0.5d0
783  real(real64), parameter :: three = 3.0d0
784  real(real64), parameter :: six = 6.0d0
785 
786  ! Local Variables
787  integer(int32) :: right
788  real(real64) :: dt, h
789 
790  ! Initialization
791  right = jlo + 1
792  dt = pt - this%m_x(jlo)
793  h = this%m_x(right) - this%m_x(jlo)
794 
795  ! Process
796  yy = this%m_y(jlo) + dt * ((this%m_y(right) - this%m_y(jlo)) / h - &
797  (this%m_ypp(right) / six + this%m_ypp(jlo) / three) * h + &
798  dt * (half * this%m_ypp(jlo) + &
799  dt * ((this%m_ypp(right) - this%m_ypp(jlo)) / (six * h))))

842  subroutine si_second_deriv(this, ibcbeg, ybcbeg, ibcend, ybcend, err)
843  ! Arguments
844  class(spline_interp), intent(inout) :: this
845  integer(int32), intent(in) :: ibcbeg, ibcend
846  real(real64), intent(in) :: ybcbeg, ybcend
847  class(errors), intent(inout), optional, target :: err
848 
849  ! Parameters
850  real(real64), parameter :: zero = 0.0d0
851  real(real64), parameter :: one = 1.0d0
852  real(real64), parameter :: three = 3.0d0
853  real(real64), parameter :: six = 6.0d0
854 
855  ! Local Variables
856  class(errors), pointer :: errmgr
857  type(errors), target :: deferr
858  integer(int32) :: i, n, flag
859  real(real64), allocatable, dimension(:) :: a1, a2, a3, a4, a5, b
860 
861  ! Initialization
862  if (present(err)) then
863  errmgr => err
864  else
865  errmgr => deferr
866  end if
867  n = size(this%m_x)
868 
869  ! Allocate Memory
870  if (allocated(this%m_ypp)) deallocate(this%m_ypp)
871  allocate(this%m_ypp(n), stat = flag)
872  if (flag == 0) allocate(a1(n), stat = flag)
873  if (flag == 0) allocate(a2(n), stat = flag)
874  if (flag == 0) allocate(a3(n), stat = flag)
875  if (flag == 0) allocate(a4(n), stat = flag)
876  if (flag == 0) allocate(a5(n), stat = flag)
877  if (flag == 0) allocate(b(n), stat = flag)
878  if (flag /= 0) then
879  call errmgr%report_error("si_second_deriv", &
880  "Insufficient memory available.", cf_out_of_memory_error)
881  return
882  end if
883 
884  ! Zero out the matrix
885  do i = 1, n
886  a1(i) = zero
887  a2(i) = zero
888  a3(i) = zero
889  a4(i) = zero
890  a5(i) = zero
891  end do
892 
893  ! Set the first equation
894  select case (ibcbeg)
895  case (spline_quadratic_over_interval)
896  b(1) = zero
897  a3(1) = one
898  a4(1) = -one
899  case (spline_known_first_derivative)
900  b(1) = (this%m_y(2) - this%m_y(1)) / &
901  (this%m_x(2) - this%m_x(1)) - ybcbeg
902  a3(1) = (this%m_x(2) - this%m_x(1)) / three
903  a4(1) = (this%m_x(2) - this%m_x(1)) / six
904  case (spline_known_second_derivative)
905  b(1) = ybcbeg
906  a3(1) = one
907  a4(1) = zero
908  case (spline_continuous_third_derivative)
909  b(1) = zero
910  a3(1) = this%m_x(2) - this%m_x(3)
911  a4(1) = this%m_x(3) - this%m_x(1)
912  a5(1) = this%m_x(1) - this%m_x(2)
913  case default
914  b(1) = zero
915  a3(1) = one
916  a4(1) = one
917  end select
918 
919  ! Set the intermediate equations
920  do i = 2, n - 1

922  (this%m_x(i+1) - this%m_x(i)) - &
923  (this%m_y(i) - this%m_y(i-1)) / (this%m_x(i) - this%m_x(i-1))
924  a2(i) = (this%m_x(i+1) - this%m_x(i)) / six
925  a3(i) = (this%m_x(i+1) - this%m_x(i-1)) / three
926  a4(i) = (this%m_x(i) - this%m_x(i-1)) / six
927  end do
928

930  select case (ibcend)
931  case (spline_quadratic_over_interval)
932  b(n) = zero
933  a2(n) = -one
934  a3(n) = one
935  case (spline_known_first_derivative)
936  b(n) = ybcend - (this%m_y(n) - this%m_y(n-1)) / &
937  (this%m_x(n) - this%m_x(n-1))

939  a3(n) = (this%m_x(n) - this%m_x(n-1)) / three
940  case (spline_known_second_derivative)
941  b(n) = ybcend
942  a2(n) = zero
943  a3(n) = one
944  case (spline_continuous_third_derivative)
945  b(n) = zero
946  a1(n) = this%m_x(n-1) - this%m_x(n)
947  a2(n) = this%m_x(n) - this%m_x(n-2)
948  a3(n) = this%m_x(n-2) - this%m_x(n-1)
949  case default
950  b(n) = zero
951  a2(n) = -one
952  a3(n) = one
953  end select
954 
955  ! Define the second derivative
956  if (n == 2 .and. ibcbeg == spline_quadratic_over_interval .and. &
957  ibcend == spline_quadratic_over_interval) then
958  ! Deal with the special case of N = 2, and IBCBEG = IBCEND = 0
959  this%m_ypp(1) = zero
960  this%m_ypp(2) = zero
961  else
962  ! Solve the linear system
963  call penta_solve(a1, a2, a3, a4, a5, b, this%m_ypp)
964  end if
