542  function im_locate(this, pt, err) result(j)
543  ! Arguments
544  class(interp_manager), intent(inout) :: this
545  real(real64), intent(in) :: pt
546  class(errors), intent(inout), optional, target :: err
547  integer :: j
548 
549  ! Local Variables
550  integer(int32) :: n, m, jhi, jmid, jlo
551  logical :: ascnd
552  class(errors), pointer :: errmgr
553  type(errors), target :: deferr
554 
555  ! Initialization
556  j = 0
557  n = size(this%m_x)
558  m = this%m_order + 1
559  ascnd = this%m_x(n) >= this%m_x(1)
560  jlo = 1
561  jhi = n
562  if (present(err)) then
563  errmgr => err
564  else
565  errmgr => deferr
566  end if
567 
568  ! Ensure data has been defined
569  if (.not.allocated(this%m_x) .or. .not.allocated(this%m_y)) then
570  call errmgr%report_error("im_locate", "No data has been defined.", &
571  cf_no_data_defined_error)
572  return
573  end if
574 
575  ! Process
576  do while (jhi - jlo > 1)
577  jmid = (jhi + jlo) / 2
578  if (pt >= this%m_x(jmid) .eqv. ascnd) then
579  jlo = jmid
580  else
581  jhi = jmid
582  end if
583  end do
584 
585  ! Check to see if we should use a more efficient search approach next
586  ! time
587  this%m_correlated = abs(jlo - this%m_savedIndex) <= this%m_indexCheck
588  this%m_savedIndex = jlo
589 
590  ! Output
591  ! j = max(1, min(n + 1 - m, jlo - (m - 1) / 2))
592  if (pt == this%m_x(1)) then
593  j = 1
594  else if (pt == this%m_x(n)) then
595  j = n - 1
596  else
597  j = jlo
598  end if

617  function im_hunt(this, pt, err) result(j)
618  ! Arguments
619  class(interp_manager), intent(inout) :: this
620  real(real64), intent(in) :: pt
621  class(errors), intent(inout), optional, target :: err
622  integer(int32) :: j
623 
624  ! Local Variables
625  integer(int32) :: jlo, jmid, jhi, inc, n, m
626  logical :: ascnd
627  class(errors), pointer :: errmgr
628  type(errors), target :: deferr
629 
630 
631  ! Initialization
632  j = 0
633  n = size(this%m_x)
634  m = this%m_order + 1
635  jlo = this%m_savedIndex
636  inc = 1
637  ascnd = this%m_x(n) > this%m_x(1)
638  if (present(err)) then
639  errmgr => err
640  else
641  errmgr => deferr
642  end if
643 
644  ! Ensure data has been defined
645  if (.not.allocated(this%m_x) .or. .not.allocated(this%m_y)) then
646  call errmgr%report_error("im_hunt", "No data has been defined.", &
647  cf_no_data_defined_error)
648  return
649  end if
650 
651  ! Process
652  if (jlo < 1 .or. jlo > n) then
653  jlo = 1
654  jhi = n
655  else
656  if (pt >= this%m_x(jlo) .eqv. ascnd) then
657  do
658  jhi = jlo + inc
659  if (jhi >= n) then
660  jhi = n
661  exit
662  else if (pt < this%m_x(jhi) .eqv. ascnd) then
663  exit
664  else
665  jlo = jhi
666  inc = inc + inc
667  end if
668  end do
669  else
670  jhi = jlo
671  do
672  jlo = jlo - inc
673  if (jlo <= 1) then
674  jlo = 1
675  exit
676  else if (pt >= this%m_x(jlo) .eqv. ascnd) then
677  exit
678  else
679  jhi = jlo
680  inc = inc + inc
681  end if
682  end do
683  end if
684  end if
685 
686  ! The hunt is done, so begin the final bisection phase
687  do while (jhi - jlo > 1)
688  jmid = (jhi + jlo) / 2
689  if (pt >= this%m_x(jmid) .eqv. ascnd) then
690  jlo = jmid
691  else
692  jhi = jmid
693  end if
694  end do
695 
696  ! Check to see if we should hunt or locate the next time around
697  this%m_correlated = abs(jlo - this%m_savedIndex) <= this%m_indexCheck
698  this%m_savedIndex = jlo
699 
700  ! Output
701  ! j = max(1, min(n + 1 - m, jlo - (m - 1) / 2))
702  if (pt == this%m_x(1)) then
703  j = 1
704  else if (pt == this%m_x(n)) then
705  j = n - 1
706  else
707  j = jlo
708  end if
709  end function

929  function pi_raw_interp(this, jlo, pt) result(yy)
930  ! Arguments
931  class(polynomial_interp), intent(inout) :: this
932  integer(int32), intent(in) :: jlo
933  real(real64), intent(in) :: pt
934  real(real64) :: yy
935 
936  ! Local Variables
937  integer(int32) :: i, ind, m, ns, mm, jl
938  real(real64) :: den, dif, dift, ho, hp, w
939 
940  ! Initialization
941  mm = this%m_order + 1
942  ns = 1
943  jl = jlo - 1
944  dif = abs(pt - this%m_x(jl))
945 
946  ! Find the index NS of the closest table entry, and then initialize
947  ! the C and D arrays.
948  do i = 1, mm
949  ind = jl + i - 1
950  dift = abs(pt - this%m_x(ind))
951  if (dift < dif) then
952  ns = i
953  dif = dift
954  end if
955  this%m_c(i) = this%m_y(ind)
956  this%m_d(i) = this%m_y(ind)
957  end do
958 
959  ! Define the initial approximation to the interpolated point
960  yy = this%m_y(jl + ns - 1)
961  ns = ns - 1
962 
963  ! Build the tables, and define the interpolated point
964  do m = 1, mm-1
965  do i = 1, mm - m
966  ind = jl + i - 1
967  ho = this%m_x(ind) - pt
968  hp = this%m_x(ind+m) - pt
969  w = this%m_c(i+1) - this%m_d(i)
970  den = ho - hp
971  den = w / den
972  this%m_d(i) = hp * den
973  this%m_c(i) = ho * den
974  end do
975  if (2 * ns < mm - m) then
976  this%m_dy = this%m_c(ns + 1)
977  else
978  this%m_dy = this%m_d(ns)
979  ns = ns - 1
980  end if
981  yy = yy + this%m_dy
982  end do
983  end function

1012  subroutine penta_solve(a1, a2, a3, a4, a5, b, x)
1013  ! Arguments
1014  real(real64), intent(in), dimension(:) :: a1, a5
1015  real(real64), intent(inout), dimension(:) :: a2, a3, a4, b
1016  real(real64), intent(out), dimension(:) :: x
1017 
1018  ! Local Variables
1019  integer(int32) :: i, n
1020  real(real64) :: xmult
1021 
1022  ! Initialization
1023  n = size(a1)
1024 
1025  ! Process
1026  do i = 2, n - 1
1027  xmult = a2(i) / a3(i - 1)
1028  a3(i) = a3(i) - xmult * a4(i - 1)
1029  a4(i) = a4(i) - xmult * a5(i - 1)
1030  b(i) = b(i) - xmult * b(i - 1)
1031  xmult = a1(i + 1) - xmult * a4(i - 1)
1032  a2(i + 1) = a2(i + 1) - xmult * a4(i - 1)
1033  a3(i + 1) = a3(i + 1) - xmult * a5(i - 1)
1034  b(i + 1) = b(i + 1) - xmult * b(i - 1)
1035  end do
1036 
1037  xmult = a2(n) / a3(n - 1)
1038  a3(n) = a3(n) - xmult * a4(n - 1)
1039  x(n) = (b(n) - xmult * b(n - 1)) / a3(n)
1040  x(n - 1) = (b(n - 1) - a4(n - 1) * x(n)) / a3(n - 1)
1041  do i = n - 2, 1, -1
1042  x(i) = (b(i) - a4(i) * x(i + 1) - a5(i) * x(i + 2)) / a3(i)
1043  end do
1044  end subroutine

1122  subroutine si_second_deriv(this, ibcbeg, ybcbeg, ibcend, ybcend, err)
1123  ! Arguments
1124  class(spline_interp), intent(inout) :: this
1125  integer(int32), intent(in) :: ibcbeg, ibcend
1126  real(real64), intent(in) :: ybcbeg, ybcend
1127  class(errors), intent(inout), optional, target :: err
1128 
1129  ! Parameters
1130  real(real64), parameter :: zero = 0.0d0
1131  real(real64), parameter :: one = 1.0d0
1132  real(real64), parameter :: three = 3.0d0
1133  real(real64), parameter :: six = 6.0d0
1134 
1135  ! Local Variables
1136  class(errors), pointer :: errmgr
1137  type(errors), target :: deferr
1138  integer(int32) :: i, n, flag
1139  real(real64), allocatable, dimension(:) :: a1, a2, a3, a4, a5, b
1140 
1141  ! Initialization
1142  if (present(err)) then
1143  errmgr => err
1144  else
1145  errmgr => deferr
1146  end if
1147  n = size(this%m_x)
1148 
1149  ! Allocate Memory
1150  if (allocated(this%m_ypp)) deallocate(this%m_ypp)
1151  allocate(this%m_ypp(n), stat = flag)
1152  if (flag == 0) allocate(a1(n), stat = flag)
1153  if (flag == 0) allocate(a2(n), stat = flag)
1154  if (flag == 0) allocate(a3(n), stat = flag)
1155  if (flag == 0) allocate(a4(n), stat = flag)
1156  if (flag == 0) allocate(a5(n), stat = flag)
1157  if (flag == 0) allocate(b(n), stat = flag)
1158  if (flag /= 0) then
1159  call errmgr%report_error("si_second_deriv", &
1160  "Insufficient memory available.", cf_out_of_memory_error)
1161  return
1162  end if
1163 
1164  ! Zero out the matrix
1165  do i = 1, n
1166  a1(i) = zero
1167  a2(i) = zero
1168  a3(i) = zero
1169  a4(i) = zero
1170  a5(i) = zero
1171  end do
1172 
1173  ! Set the first equation
1174  select case (ibcbeg)
1175  case (spline_quadratic_over_interval)
1176  b(1) = zero
1177  a3(1) = one
1178  a4(1) = -one
1179  case (spline_known_first_derivative)
1180  b(1) = (this%m_y(2) - this%m_y(1)) / &
1181  (this%m_x(2) - this%m_x(1)) - ybcbeg
1182  a3(1) = (this%m_x(2) - this%m_x(1)) / three
1183  a4(1) = (this%m_x(2) - this%m_x(1)) / six
1184  case (spline_known_second_derivative)
1185  b(1) = ybcbeg
1186  a3(1) = one
