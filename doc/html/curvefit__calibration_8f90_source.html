580  real(real64), intent(in), dimension(:) :: applied, measured

582  class(errors), intent(inout), optional, target :: err
583  real(real64) :: rst
584 
585  ! Parameters
586  real(real64), parameter :: zero = 0.0d0
587 
588  ! Local Variables
589  integer(int32) :: i, i1, i2, n
590  real(real64) :: t, eps
591  class(errors), pointer :: errmgr
592  type(errors), target :: deferr
593 
594  ! Initialization
595  n = size(applied)
596  rst = zero
597  eps = epsilon(eps)
598  t = zero
599  if (present(tol)) t = tol
600  if (t <= eps) then
601  i1 = idamax(n, applied, 1)
602  t = sqrt(eps) * abs(applied(i1))
603  end if
604  if (present(err)) then
605  errmgr => err
606  else
607  errmgr => deferr
608  end if
609 
610  ! Input Checking
611  if (size(measured) /= n) then
612  call errmgr%report_error("rtz_1", "The measured data " // &
613  "array must be the same size as the applied data array.", &
614  cf_array_size_error)
615  return
616  end if
617 
618  ! Determine the indices of the matching values - base the search on the
619  ! applied values, not the measured values
620  i1 = 0
621  i2 = 0
622  do i = 1, n
623  if (abs(applied(i)) < t) then
624  i1 = i
625  exit
626  end if
627  end do
628  if (i1 == 0) return
629  do i = i1 + 1, n
630  if (abs(applied(i)) < t) then
631  i2 = i
632  exit
633  end if
634  end do
635  if (i2 == 0) return
636 
637  ! Compute the return to zero error
638  rst = measured(i2) - measured(i1)
639  end function

664  function repeat_1(applied, measured, err) result(rst)
665  ! Arguments
666  real(real64), intent(in), dimension(:,:) :: applied, measured
667  class(errors), intent(inout), optional, target :: err
668  real(real64) :: rst
669 
670  ! Parameters
671  real(real64), parameter :: zero = 0.0d0
672 
673  ! Local Variables
674  integer(int32) :: i, npts, ntests, ip
675  class(errors), pointer :: errmgr
676  type(errors), target :: deferr
677  type(linear_interp) :: interp
678  logical :: ascending
679  real(real64), allocatable, dimension(:) :: y
680 
681  ! Initialization
682  npts = size(applied, 1)
683  ntests = size(applied, 2)
684  rst = zero
685  if (present(err)) then
686  errmgr => err
687  else
688  errmgr => deferr
689  end if
690 
691  ! Input Check
692  if (size(measured, 1) /= npts .or. size(measured, 2) /= ntests) then
693  call errmgr%report_error("repeat_1", "The measured data array " // &
694  "must be the same size as the applied data array.", &
695  cf_array_size_error)
696  return
697  end if
698 
699  ! Quick Return
700  if (npts < 2 .or. ntests < 2) return
701 
702  ! Determine where the inflection point is within the data such that
703  ! ascending and descending segments may be considered seperately
704  ascending = applied(2,1) - applied(1,1) > zero
705  ip = npts
706  do i = 2, npts
707  if (ascending) then
708  if (applied(i,1) < applied(i-1,1)) then
709  ip = i - 1
710  exit
711  end if
712  else
713  if (applied(i,1) > applied(i-1,1)) then
714  ip = i - 1
715  exit
716  end if
717  end if
718  end do
719 
720  ! Cycle over each test, and use the initial test (column 1) as the
721  ! reference
722  do i = 2, ntests
723  ! Determine how far subsequent tests move from the initial test
724  ! for the ascending data
725  call interp%initialize(applied(1:ip,i), measured(1:ip,i), &
726  err = errmgr)
727  if (errmgr%has_error_occurred()) return
728  y = abs(interp%interpolate(applied(1:ip,1)) - measured(1:ip,1))
729  rst = max(rst, maxval(y))
730 
731  ! Consider the descending data
732  if (ip < npts) then
733  call interp%initialize(applied(ip:npts,i), &
734  measured(ip:npts,i), err = errmgr)
735  if (errmgr%has_error_occurred()) return
736  y = abs(interp%interpolate(applied(ip:npts,1)) - &
737  measured(ip:npts,1))
738  rst = max(rst, maxval(y))
739  end if
740  end do
741  end function

941  function xtalk_1(xerr, indices, err) result(xt)
942  ! Arguments
943  real(real64), intent(in), dimension(:,:) :: xerr
944  integer(int32), intent(in), dimension(:) :: indices
945  real(real64), dimension(size(xerr, 2), size(xerr, 2)) :: xt
946  class(errors), intent(inout), optional, target :: err
947 
948  ! Parameters
949  real(real64), parameter :: zero = 0.0d0
950 
951  ! Local Variables
952  integer(int32) :: i, j, n, npts, ndof, first, last, ind
953  class(errors), pointer :: errmgr
954  type(errors), target :: deferr
955 
956  ! Initialization
957  npts = size(xerr, 1)
958  ndof = size(xerr, 2)
959  n = size(indices) ! must be even
960  if (present(err)) then
961  errmgr => err
962  else
963  errmgr => deferr
964  end if
965  xt = zero
966 
967  ! Input Check
968  if (size(indices) /= 2 * ndof) then
969  call errmgr%report_error("xtalk_1", "The indices array is " // &
970  "not compatible in size with the supplied data array.", &
971  cf_array_size_error)
972  return
973  end if
974  if (maxval(indices) > npts .or. minval(indices) < 1) then
975  call errmgr%report_error("xtalk_1", "There are entries in " // &
976  "indices that exceed the bounds of the input data array.", &
977  cf_array_index_error)
978  return
979  end if
980 
981 
982  ! Process
983  do i = 1, ndof ! Cycle over each DOF
984  ! Locate the starting and stopping row indices for the DOF
985  first = indices(2 * i - 1)
986  last = indices(2 * i)
987 
988  ! Compute the error for each DOF over the specified range
989  do j = 1, ndof
990  if (i /= j) then
991  npts = last - first + 1
992  ind = idamax(npts, xerr(first:last,j), 1) + first - 1
993  xt(i,j) = xerr(ind,j)
994  else
995  xt(i,j) = zero
996  end if
997  end do
998  end do

1066  subroutine split_ascend_descend_1(x, ascend, descend, nascend, ndescend, &
1067  err)
1068  ! Arguments
1069  real(real64), intent(in), dimension(:) :: x
1070  real(real64), intent(out), dimension(:) :: ascend, descend
1071  integer(int32), intent(out) :: nascend, ndescend
1072  class(errors), intent(inout), optional, target :: err
1073 
1074  ! Parameters
1075  real(real64), parameter :: zero = 0.0d0
1076  real(real64), parameter :: tol = 0.5d-2
1077 
1078  ! Local Variables
1079  logical :: is_ascend, check
1080  integer(int32) :: i, i1, i2, n
1081  real(real64) :: t
1082  class(errors), pointer :: errmgr
1083  type(errors), target :: deferr
1084  character(len = 256) :: errmsg
1085 
1086  ! Initialization
1087  n = size(x)
1088  nascend = 0
1089  ndescend = 0
1090  if (n >= 2) then
1091  is_ascend = x(2) - x(1) > zero
1092  else
1093  is_ascend = .false.
1094  end if
1095  if (present(err)) then
1096  errmgr => err
1097  else
1098  errmgr => deferr
1099  end if
1100 
1101  ! Determine where the data is no longer monotonic
1102  do i = 1, n
1103  if (i < n) then
1104  check = x(i+1) - x(i) < zero
1105  else
1106  check = is_ascend
1107  end if
1108  if (check .eqv. is_ascend) then
1109  nascend = i
1110  exit
1111  end if
1112  end do
1113 
1114  ! Ensure the receiving array is appropriately sized, and then copy over
1115  ! the correct items from X.
1116  if (size(ascend) < nascend) then
1117  write(errmsg, '(AI0AI0A)') "The receiving array for the " // &
1118  "ascending data set is too small. The minimum required " // &
1119  "dimension is: ", nascend, ", but the current size is: ", &
1120  size(ascend), "."
1121  call errmgr%report_error("split_ascend_descend_1", trim(errmsg), &
1122  cf_array_size_error)
1123  return
1124  end if
1125  ascend(1:nascend) = x(1:nascend)
1126 
1127  ! If NASCEND == N, then the data set only included ascending data.
1128  if (nascend == n) then
1129  return
1130  end if
1131 
1132  ! As we now have the apex of the data set, scale the tolerance value
1133  ! based upon the largest magnitude value. The tolerance value will be
1134  ! used to determine when we've returned back to the original starting
1135  ! point (useful for data sets that cross zero).
1136  t = tol * abs(x(nascend))
1137 
1138  ! Assuming we're here, it's now safe to start the descending search.
1139  ! Search until either we get to the same value, within T, of where we
1140  ! started, or the end of the data set. Whichever comes first. The
1141  ! other check is to determine if there is a switch in direction in the
1142  ! data.
1143  !i1 = nascend + 1
1144  i1 = nascend
1145  i2 = i1
1146  do i = i1, n
1147  if (i < n) then
1148  check = x(i+1) - x(i) > zero
1149  else
1150  check = .not.is_ascend
1151  end if
1152  if (abs(x(i) - x(1)) < t .or. check .eqv. is_ascend) then
1153  i2 = i
1154  exit
1155  end if
1156  end do
1157  if (i2 == i1) i2 = n
1158  ndescend = i2 - i1 + 1
1159 
1160  ! Ensure DESCEND is appropriately sized
1161  if (size(descend) < ndescend) then
1162  write(errmsg, '(AI0AI0A)') "The receiving array for the " // &
1163  "descending data set is too small. The minimum required " // &
1164  "dimension is: ", nascend, ", but the current size is: ", &
1165  size(ascend), "."
1166  call errmgr%report_error("split_ascend_descend_1", trim(errmsg), &
1167  cf_array_size_error)
1168  return
1169  end if
1170  descend(1:ndescend) = x(i1:i2)
1171  end subroutine
