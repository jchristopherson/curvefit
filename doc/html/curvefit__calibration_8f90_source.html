143  ! Initialization
144  eps = epsilon(eps)

147  errmgr => err

151

154  call errmgr%report_error("seb_1", "The output data array " // &
155  "must be the same size as the applied data array.", &

176  a = abs((output(j) - s * ratio(j)) / s)
177  else
178  s = zero
179  a = zero
180  end if
181  if (a >= b) then
182  b = a
183  t = s
184  end if
185  end if
186  end do
187  end do

191  rst%output = t

524  real(real64), intent(in), dimension(:) :: applied, measured

526  class(errors), intent(inout), optional, target :: err
527  real(real64) :: rst
528 
529  ! Parameters
530  real(real64), parameter :: zero = 0.0d0
531 
532  ! Local Variables
533  integer(int32) :: i, i1, i2, n
534  real(real64) :: t, eps
535  class(errors), pointer :: errmgr
536  type(errors), target :: deferr
537 
538  ! Initialization
539  n = size(applied)
540  rst = zero
541  eps = epsilon(eps)
542  t = zero
543  if (present(tol)) t = tol
544  if (t <= eps) then
545  i1 = idamax(n, applied, 1)
546  t = sqrt(eps) * abs(applied(i1))
547  end if
548  if (present(err)) then
549  errmgr => err
550  else
551  errmgr => deferr
552  end if
553 
554  ! Input Checking
555  if (size(measured) /= n) then
556  call errmgr%report_error("rtz_1", "The measured data " // &
557  "array must be the same size as the applied data array.", &
558  cf_array_size_error)
559  return
560  end if
561 
562  ! Determine the indices of the matching values - base the search on the
563  ! applied values, not the measured values
564  i1 = 0
565  i2 = 0
566  do i = 1, n
567  if (abs(applied(i)) < t) then
568  i1 = i
569  exit
570  end if
571  end do
572  if (i1 == 0) return
573  do i = i1 + 1, n
574  if (abs(applied(i)) < t) then
575  i2 = i
576  exit
577  end if
578  end do
579  if (i2 == 0) return
580 
581  ! Compute the return to zero error
582  rst = measured(i2) - measured(i1)
583  end function

608  function repeat_1(applied, measured, err) result(rst)
609  ! Arguments
610  real(real64), intent(in), dimension(:,:) :: applied, measured
611  class(errors), intent(inout), optional, target :: err
612  real(real64) :: rst
613 
614  ! Parameters
615  real(real64), parameter :: zero = 0.0d0
616 
617  ! Local Variables
618  integer(int32) :: i, npts, ntests, ip
619  class(errors), pointer :: errmgr
620  type(errors), target :: deferr
621  type(linear_interp) :: interp
622  logical :: ascending
623  real(real64), allocatable, dimension(:) :: y
624 
625  ! Initialization
626  npts = size(applied, 1)
627  ntests = size(applied, 2)
628  rst = zero
629  if (present(err)) then
630  errmgr => err
631  else
632  errmgr => deferr
633  end if
634 
635  ! Input Check
636  if (size(measured, 1) /= npts .or. size(measured, 2) /= ntests) then
637  call errmgr%report_error("repeat_1", "The measured data array " // &
638  "must be the same size as the applied data array.", &
639  cf_array_size_error)
640  return
641  end if
642 
643  ! Quick Return
644  if (npts < 2 .or. ntests < 2) return
645 
646  ! Determine where the inflection point is within the data such that
647  ! ascending and descending segments may be considered seperately
648  ascending = applied(2,1) - applied(1,1) > zero
649  ip = npts
650  do i = 2, npts
651  if (ascending) then
652  if (applied(i,1) < applied(i-1,1)) then
653  ip = i - 1
654  exit
655  end if
656  else
657  if (applied(i,1) > applied(i-1,1)) then
658  ip = i - 1
659  exit
660  end if
661  end if
662  end do
663 
664  ! Cycle over each test, and use the initial test (column 1) as the
665  ! reference
666  do i = 2, ntests
667  ! Determine how far subsequent tests move from the initial test
668  ! for the ascending data
669  call interp%initialize(applied(1:ip,i), measured(1:ip,i), &
670  err = errmgr)
671  if (errmgr%has_error_occurred()) return
672  y = abs(interp%interpolate(applied(1:ip,1)) - measured(1:ip,1))
673  rst = max(rst, maxval(y))
674 
675  ! Consider the descending data
676  if (ip < npts) then
677  call interp%initialize(applied(ip:npts,i), &
678  measured(ip:npts,i), err = errmgr)
679  if (errmgr%has_error_occurred()) return
680  y = abs(interp%interpolate(applied(ip:npts,1)) - &
681  measured(ip:npts,1))
682  rst = max(rst, maxval(y))
683  end if
684  end do
685  end function

1010  subroutine split_ascend_descend_1(x, ascend, descend, nascend, ndescend, &
1011  err)
1012  ! Arguments
1013  real(real64), intent(in), dimension(:) :: x
1014  real(real64), intent(out), dimension(:) :: ascend, descend
1015  integer(int32), intent(out) :: nascend, ndescend
1016  class(errors), intent(inout), optional, target :: err
1017 
1018  ! Parameters
1019  real(real64), parameter :: zero = 0.0d0
1020  real(real64), parameter :: tol = 0.5d-2
1021 
1022  ! Local Variables
1023  logical :: is_ascend, check
1024  integer(int32) :: i, i1, i2, n
1025  real(real64) :: t
1026  class(errors), pointer :: errmgr
1027  type(errors), target :: deferr
1028  character(len = 256) :: errmsg
1029 
1030  ! Initialization
1031  n = size(x)
1032  nascend = 0
1033  ndescend = 0
1034  if (n >= 2) then
1035  is_ascend = x(2) - x(1) > zero
1036  else
1037  is_ascend = .false.
1038  end if
1039  if (present(err)) then
1040  errmgr => err
1041  else
1042  errmgr => deferr
1043  end if
1044 
1045  ! Determine where the data is no longer monotonic
1046  do i = 1, n
1047  if (i < n) then
1048  check = x(i+1) - x(i) < zero
1049  else
1050  check = is_ascend
1051  end if
1052  if (check .eqv. is_ascend) then
1053  nascend = i
1054  exit
1055  end if
1056  end do
1057 
1058  ! Ensure the receiving array is appropriately sized, and then copy over
1059  ! the correct items from X.
1060  if (size(ascend) < nascend) then
1061  write(errmsg, '(AI0AI0A)') "The receiving array for the " // &
1062  "ascending data set is too small. The minimum required " // &
1063  "dimension is: ", nascend, ", but the current size is: ", &
1064  size(ascend), "."
1065  call errmgr%report_error("split_ascend_descend_1", trim(errmsg), &
1066  cf_array_size_error)
1067  return
1068  end if
1069  ascend(1:nascend) = x(1:nascend)
1070 
1071  ! If NASCEND == N, then the data set only included ascending data.
1072  if (nascend == n) then
1073  return
1074  end if
1075 
1076  ! As we now have the apex of the data set, scale the tolerance value
1077  ! based upon the largest magnitude value. The tolerance value will be
1078  ! used to determine when we've returned back to the original starting
1079  ! point (useful for data sets that cross zero).
1080  t = tol * abs(x(nascend))
1081 
1082  ! Assuming we're here, it's now safe to start the descending search.
1083  ! Search until either we get to the same value, within T, of where we
1084  ! started, or the end of the data set. Whichever comes first. The
1085  ! other check is to determine if there is a switch in direction in the
1086  ! data.
1087  !i1 = nascend + 1
1088  i1 = nascend
1089  i2 = i1
1090  do i = i1, n
1091  if (i < n) then
1092  check = x(i+1) - x(i) > zero
1093  else
1094  check = .not.is_ascend
1095  end if
1096  if (abs(x(i) - x(1)) < t .or. check .eqv. is_ascend) then
1097  i2 = i
1098  exit
1099  end if
1100  end do
1101  if (i2 == i1) i2 = n
1102  ndescend = i2 - i1 + 1
1103 
1104  ! Ensure DESCEND is appropriately sized
1105  if (size(descend) < ndescend) then
1106  write(errmsg, '(AI0AI0A)') "The receiving array for the " // &
1107  "descending data set is too small. The minimum required " // &
1108  "dimension is: ", nascend, ", but the current size is: ", &
1109  size(ascend), "."
1110  call errmgr%report_error("split_ascend_descend_1", trim(errmsg), &
1111  cf_array_size_error)
1112  return
1113  end if
1114  descend(1:ndescend) = x(i1:i2)
1115  end subroutine
